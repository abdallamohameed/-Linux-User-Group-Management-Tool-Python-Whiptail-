#!/usr/bin/env python3
import os
import subprocess

# ----------------------------------------------
def menu(title, items, h="22", w="75", lh="13"):
    """
    items = [("Tag","Desc"), ...]
    returns Tag or None if exit/esc
    """
    cmd = ["whiptail", "--title", title, "--menu", "", h, w, lh]
    for t, d in items:
        cmd += [t, d]

    p = subprocess.run(cmd, stderr=subprocess.PIPE, text=True)
    if p.returncode != 0:
        return None
    return p.stderr.strip()

def ask(title, text, h="10", w="60"):
    p = subprocess.run(
        ["whiptail", "--title", title, "--inputbox", text, h, w, ""],
        stderr=subprocess.PIPE,
        text=True
    )
    if p.returncode != 0:
        return None
    return p.stderr.strip()

def msg(title, text, h="12", w="70"):
    subprocess.run(["whiptail", "--title", title, "--msgbox", text, h, w])

def yesno(title, text, h="10", w="60"):
    p = subprocess.run(["whiptail", "--title", title, "--yesno", text, h, w])
    return p.returncode == 0

# ------------------------------------------------
def user_exists(username):
    return os.system(f"id {username} >/dev/null 2>&1") == 0

def group_exists(groupname):
    return os.system(f"getent group {groupname} >/dev/null 2>&1") == 0
#----------------------------------------------------------------
def add_user_wizard():

    username = ask("Add User", "Enter username:")
    if not username:
        return
    username = username.strip()

    if user_exists(username):
        msg("Error", "User already exists.")
        return

    comment = ask("Add User", "Comment -c (optional):")
    if comment is None:
        return

    with_home = yesno("Add User", "Create home directory? (-m)")

    shell = ask("Add User", "Shell (optional, example /bin/bash):")
    if shell is None:
        return

    # ---- Create user ----
    cmd = ["useradd"]

    if with_home:
        cmd += ["-m"]
    if comment:
        cmd += ["-c", comment]
    if shell:
        cmd += ["-s", shell]

    cmd += [username]

    subprocess.run(cmd)


    if yesno("Add User", "Set password now?"):
        os.system(f"passwd {username}")
        msg("Done", "Password set.")
        # 6) Final Options
    while True:
        choice = menu("Final Options",
                      [
                          ("1", "Set primary group (-g)"),
                          ("2", "Add groups (-aG)"),
                          ("3", "Set UID (-u)"),
                          ("0", "Finish")
                      ],
                      h="18", w="70", lh="8")

        if choice in (None, "0"):
            msg("Finished", "Add User completed.")
            return

        elif choice == "1":
            g = ask("Primary Group", "Enter group name:")
            if not g:
                continue

            g = g.strip()
            if not group_exists(g):
                msg("Error", "Group does not exist.")
                continue

            os.system(f"usermod -g {g} {username}")
            msg("Done", "Primary group updated.")

        elif choice == "2":
            gs = ask("Groups", "Enter groups (comma separated):")
            if not gs:
                continue

            gs = gs.strip()
            groups = gs.split(",")

            error = False
            for g in groups:
                if not group_exists(g.strip()):
                    msg("Error", f"Group '{g}' does not exist.")
                    error = True
                    break

            if error:
                continue

            os.system(f"usermod -aG {gs} {username}")
            msg("Done", "Groups added.")

        elif choice == "3":
            uid = ask("UID", "Enter UID number:")
            if not uid:
                continue

            uid = uid.strip()

            if os.system(f"getent passwd {uid} >/dev/null 2>&1") == 0:
                msg("Error", "This UID is already used.")
                continue

            os.system(f"usermod -u {uid} {username}")
            msg("Done", "UID updated.")

def delete_group():
    g = ask("Delete Group", "Enter group name to delete:")
    if not g:
        msg("Error", "Group  name is required.")
        return

    g = g.strip()

    if not group_exists(g):
        msg("Error", "Group does not exist.")
        return
    # get group gid
    gid = os.popen(f"getent group {g} | cut -d: -f3").read().strip()

    # check if used as primary group
    check = os.system(f"awk -F: '$4=={gid} {{exit 1}}' /etc/passwd")

    if check != 0:
        msg("Error", "This group is a primary group for a user.\nCannot delete it.")
        return

    # delete group
    os.system(f"groupdel {g}")

    msg("Done", "Group deleted successfully.")
